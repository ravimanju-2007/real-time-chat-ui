/* =====================================
   CONNECTION STATUS
===================================== */

function setStatus(status) {

    connectionIndicator.className = "status";

    switch(status){

        case "connected":

            connectionIndicator.classList.add("connected");

            connectionText.textContent =
              "Connected";

            break;

        case "disconnected":

            connectionIndicator.classList.add("disconnected");

            connectionText.textContent =
              "Disconnected";

            break;

        case "reconnecting":

            connectionIndicator.classList.add("reconnecting");

            connectionText.textContent =
              "Reconnecting...";

            break;
    }
}

/* =====================================
   WEBSOCKET MANAGER
===================================== */

class WSManager {

    constructor(url){

        this.url = url;

        this.connect();
    }

    connect(){

        setStatus("reconnecting");

        ws = new WebSocket(this.url);

        ws.onopen = () => {

            setStatus("connected");

            reconnectDelay = 1000;

            showToast("Connected");

            this.flushQueue();
        };

        ws.onmessage = (event) => {

            let data;

            try {

                data = JSON.parse(event.data);

            } catch {

                data = {
                    text:event.data
                };
            }

            receiveMessage(data);
        };

        ws.onerror = () => {

            ws.close();
        };

        ws.onclose = () => {

            setStatus("disconnected");

            showToast(
              `Reconnecting in ${
                  reconnectDelay/1000
              }s`
            );

            setTimeout(() => {

                this.connect();

            }, reconnectDelay);

            reconnectDelay =
              Math.min(
                reconnectDelay * 2,
                30000
              );
        };
    }

    send(data){

        const payload =
          JSON.stringify(data);

        if(
            ws &&
            ws.readyState === WebSocket.OPEN
        ){

            ws.send(payload);

        }else{

            pendingQueue.push(payload);

            showToast(
              "Message queued"
            );
        }
    }

    flushQueue(){

        while(
            pendingQueue.length > 0
        ){

            ws.send(
              pendingQueue.shift()
            );
        }
    }
}

/* =====================================
   TIME FORMAT
===================================== */

function formatTime(date){

    return new Date(date)
      .toLocaleTimeString([],{
          hour:"2-digit",
          minute:"2-digit"
      });
}

/* =====================================
   MESSAGE OBJECT
===================================== */

function createMessage(
    text,
    sender="You"
){

    return {

        id:
          Date.now() +
          Math.random(),

        text,

        sender,

        timestamp:
          new Date(),

        status:
          sender === "You"
          ? "sent"
          : "",

        self:
          sender === "You"
    };
}

/* =====================================
   RENDER MESSAGE
===================================== */

function renderMessage(message){

    const div =
      document.createElement("div");

    div.className =
      message.self
      ? "message self"
      : "message";

    div.dataset.id =
      message.id;

    div.innerHTML = `

        ${
          !message.self
          ?
          `
          <div class="avatar">
            <img src="https://i.pravatar.cc/50?img=15">
          </div>
          `
          :
          ""
        }

        <div class="message-content">

            <div class="message-header">

                <span class="sender">
                    ${message.sender}
                </span>

                <span class="timestamp">
                    ${formatTime(
                        message.timestamp
                    )}
                </span>

            </div>

            <div class="message-text">
                ${message.text}
            </div>

            ${
                message.self
                ?
                `
                <div class="message-status">

                    <span class="
                    status-icon
                    ${message.status}
                    ">
                        ${
                          message.status === "sent"
                          ? "✓"
                          : message.status === "delivered"
                          ? "✓✓"
                          : "✓✓"
                        }
                    </span>

                </div>
                `
                :
                ""
            }

        </div>
    `;

    messageList.appendChild(div);

    scrollToBottom();
}

/* =====================================
   SEND MESSAGE
===================================== */

function sendMessage(){

    const text =
      messageInput.value.trim();

    if(!text) return;

    const msg =
      createMessage(text);

    messages.push(msg);

    renderMessage(msg);

    socket.send({
        type:"message",
        text:text
    });

    updateStatus(
      msg.id,
      "delivered"
    );

    setTimeout(() => {

        updateStatus(
          msg.id,
          "read"
        );

    },2000);

    messageInput.value = "";

    updateCounter();
}

/* =====================================
   RECEIVE MESSAGE
===================================== */

function receiveMessage(data){

    const msg =
      createMessage(
        data.text ||
        "Echo message",
        "Remote User"
      );

    msg.self = false;

    messages.push(msg);

    renderMessage(msg);

    hideTyping();
}

/* =====================================
   UPDATE STATUS
===================================== */

function updateStatus(
    id,
    status
){

    const msg =
      messages.find(
        m => m.id === id
      );

    if(!msg) return;

    msg.status = status;

    const element =
      document.querySelector(
       `[data-id="${id}"] .status-icon`
      );

    if(!element) return;

    element.className =
      `status-icon ${status}`;

    if(status==="sent")
        element.textContent="✓";

    if(status==="delivered")
        element.textContent="✓✓";

    if(status==="read")
        element.textContent="✓✓";
}

/* =====================================
   RELATIVE TIME FORMAT
===================================== */

function getRelativeTime(date) {

    const seconds =
        Math.floor(
            (Date.now() - new Date(date)) / 1000
        );

    if (seconds < 60)
        return "Just now";

    const minutes =
        Math.floor(seconds / 60);

    if (minutes < 60)
        return `${minutes}m ago`;

    const hours =
        Math.floor(minutes / 60);

    if (hours < 24)
        return `${hours}h ago`;

    const days =
        Math.floor(hours / 24);

    if (days < 7)
        return `${days}d ago`;

    return new Date(date)
        .toLocaleDateString();
}

/* =====================================
   GROUPED MESSAGE RENDERING
===================================== */

function renderGroupedMessage(
    message,
    previousMessage
) {

    const wrapper =
        document.createElement("div");

    const sameSender =
        previousMessage &&
        previousMessage.sender ===
        message.sender;

    const withinFiveMinutes =
        previousMessage &&
        (
            new Date(message.timestamp)
            -
            new Date(
                previousMessage.timestamp
            )
        ) < 300000;

    const grouped =
        sameSender &&
        withinFiveMinutes;

    wrapper.className =
        message.self
        ? "message self"
        : "message";

    wrapper.dataset.id =
        message.id;

    wrapper.innerHTML = `

        ${
            !grouped && !message.self
            ?
            `
            <div class="avatar">
                <img src="https://i.pravatar.cc/50?img=15">
            </div>
            `
            :
            `
            <div style="
                width:42px;
            "></div>
            `
        }

        <div class="message-content">

            ${
                !grouped
                ?
                `
                <div class="message-header">

                    <span class="sender">
                        ${message.sender}
                    </span>

                    <span class="timestamp">
                        ${getRelativeTime(
                            message.timestamp
                        )}
                    </span>

                </div>
                `
                :
                ""
            }

            <div class="message-text">
                ${message.text}
            </div>

        </div>
    `;

    return wrapper;
}

const ITEM_HEIGHT = 90;

const BUFFER = 10;

const MAX_VISIBLE = 30;

    document.getElementById(
        "topSpacer"
    );

    document.getElementById(
        "bottomSpacer"

/* =====================================
   VIRTUAL RENDER
===================================== */

function renderVirtualMessages() {

    const scrollTop =
        chatViewport.scrollTop;

    const viewportHeight =
        chatViewport.clientHeight;

    const startIndex =
        Math.max(
            0,
            Math.floor(
                scrollTop /
                ITEM_HEIGHT
            ) - BUFFER
        );

    const visibleCount =
        Math.ceil(
            viewportHeight /
            ITEM_HEIGHT
        ) + BUFFER * 2;

    const endIndex =
        Math.min(
            messages.length,
            startIndex +
            visibleCount
        );

    messageList.innerHTML = "";

    topSpacer.style.height =
        `${startIndex * ITEM_HEIGHT}px`;

    bottomSpacer.style.height =
        `${(
            messages.length -
            endIndex
        ) * ITEM_HEIGHT}px`;

    let lastDate = null;

    for (
        let i = startIndex;
        i < endIndex;
        i++
    ) {

        const message =
            messages[i];

        const messageDate =
            new Date(
                message.timestamp
            ).toDateString();

        if (
            lastDate !== messageDate
        ) {

            messageList.appendChild(
                createDateDivider(
                    message.timestamp
                )
            );

            lastDate =
                messageDate;
        }

        const previous =
            i > 0
            ? messages[i - 1]
            : null;

        messageList.appendChild(
            renderGroupedMessage(
                message,
                previous
            )
        );
    }
}

/* =====================================
   REACTION STORAGE
===================================== */

function addReaction(
    messageId,
    emoji
) {

    const msg =
        messages.find(
            m =>
            m.id === messageId
        );

    if (!msg) return;

    if (!msg.reactions) {

        msg.reactions = {};
    }

    if (
        !msg.reactions[emoji]
    ) {

        msg.reactions[emoji] = 0;
    }

    msg.reactions[emoji]++;

    renderVirtualMessages();
}

                "image/"
            )
        ){

            msg.image =
            URL.createObjectURL(
                selectedFile
            );
        }
    }

    messages.push(msg);

    renderVirtualMessages();

    socket.send({

        type:"message",

        text:
        msg.text
    });

    messageInput.value = "";

    imagePreviewContainer
    .innerHTML = "";

    filePreviewContainer
    .innerHTML = "";

    selectedFile = null;

    updateCounter();

    scrollToBottom();
};

/* =====================================
   PATCH RENDER
===================================== */

const oldRenderGrouped =
renderGroupedMessage;

renderGroupedMessage =
function(
    message,
    previous
){

    const element =
        oldRenderGrouped(
            message,
            previous
        );

    const content =
        element.querySelector(
            ".message-content"
        );

    if (
        message.image
    ){

        const img =
        document.createElement(
            "img"
        );

        img.src =
            message.image;

        img.style.maxWidth =
            "250px";

        img.style.marginTop =
            "10px";

        img.style.borderRadius =
            "10px";

        content.appendChild(
            img
        );
    }

    if (
        message.file
    ){

        const file =
        document.createElement(
            "div"
        );

        file.className =
            "message-file";

        file.textContent =
            "📄 " +
            message.file;

        content.appendChild(
            file
        );
    }

    return element;
};

            .items;

        ){

                )
            ){

                item.getAsFile();

                file;

                    file

            }
        }
    }
);
